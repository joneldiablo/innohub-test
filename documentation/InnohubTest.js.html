  /**
   * 
   * @param {string} input string which goings to reverse
   * @returns {string} string reversed
   */
  reverse(input = '') {
    let output = [];
    for (let i = input.length; i >= 0; i--) {
      output.push(input[i]);
    }
    output = output.join('');
    return output;
  }

}
